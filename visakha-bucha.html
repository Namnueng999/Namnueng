// Google Apps Script — วิสาขบูชา ๒๕๖๙ Webhook

// Deploy → Web App → Anyone can access

const SHEET_ID = "1sA3jVT4dhzNDAYdzK9Bayqep4bd4nmLoTchTWQ8cuOU";

// POST — รับข้อมูลจากหน้าเว็บ
function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss   = SpreadsheetApp.openById(SHEET_ID);

    const sheetName = data.type === "candle" ? "เวียนเทียน"
                    : data.type === "chant"  ? "บทสวด"
                    : "ทั่วไป";

    let sheet = ss.getSheetByName(sheetName);
    if (!sheet) {
      sheet = ss.insertSheet(sheetName);
      const headers = {
        "เวียนเทียน": ["วันที่-เวลา", "ชื่อ", "คำอธิษฐาน"],
        "บทสวด":      ["วันที่-เวลา", "บทสวด", "สวดสะสม (ครั้ง)", "ชื่อผู้สวด"],
        "ทั่วไป":     ["วันที่-เวลา", "ประเภท", "ข้อมูล"],
      };
      sheet.appendRow(headers[sheetName]);
      sheet.getRange(1, 1, 1, sheet.getLastColumn())
           .setFontWeight("bold")
           .setBackground("#c9a84c")
           .setFontColor("#ffffff");
      sheet.setFrozenRows(1);
    }

    const ts = Utilities.formatDate(
      new Date(), "Asia/Bangkok", "dd/MM/yyyy HH:mm:ss"
    );

    if (data.type === "candle") {
      sheet.appendRow([ts, data.name || "", data.wish || ""]);
    } else if (data.type === "chant") {
      sheet.appendRow([ts, data.prayer || "", data.count || 0, data.name || ""]);
    } else {
      sheet.appendRow([ts, data.type || "", JSON.stringify(data)]);
    }

    // อัปเดต Summary
    _updateSummary(ss);

    return ContentService
      .createTextOutput(JSON.stringify({ ok: true }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ ok: false, error: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// GET — ดึง summary กลับมาแสดงผล
function doGet() {
  const ss  = SpreadsheetApp.openById(SHEET_ID);
  const sum = ss.getSheetByName("Summary");
  if (!sum) {
    return ContentService
      .createTextOutput(JSON.stringify({ candles: 0, chants: 0 }))
      .setMimeType(ContentService.MimeType.JSON);
  }
  const rows   = sum.getDataRange().getValues().slice(1); // skip header
  const result = {};
  rows.forEach(r => { if (r[0]) result[r[0]] = r[1]; });
  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

// ── Internal: อัปเดต Summary sheet ──
function _updateSummary(ss) {
  let sum = ss.getSheetByName("Summary");
  if (!sum) {
    sum = ss.insertSheet("Summary");
    sum.appendRow(["Key", "Value", "อัปเดตล่าสุด"]);
    sum.getRange(1,1,1,3).setFontWeight("bold").setBackground("#1c1208").setFontColor("#c9a84c");
  }

  const cs = ss.getSheetByName("เวียนเทียน");
  const ps = ss.getSheetByName("บทสวด");

  const candleCount = cs ? Math.max(0, cs.getLastRow() - 1) : 0;

  let chantTotal = 0;
  if (ps && ps.getLastRow() > 1) {
    const vals = ps.getRange(2, 3, ps.getLastRow() - 1, 1).getValues();
    chantTotal  = vals.reduce((acc, r) => acc + (Number(r[0]) || 0), 0);
  }

  const ts = Utilities.formatDate(new Date(), "Asia/Bangkok", "dd/MM/yyyy HH:mm:ss");

  // เขียนทับ data rows (ไม่แตะ header)
  if (sum.getLastRow() < 2) {
    sum.appendRow(["candles", candleCount, ts]);
    sum.appendRow(["chants",  chantTotal,  ts]);
  } else {
    sum.getRange(2, 1, 1, 3).setValues([["candles", candleCount, ts]]);
    sum.getRange(3, 1, 1, 3).setValues([["chants",  chantTotal,  ts]]);
  }
}
